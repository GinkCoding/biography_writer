### 问题 3.1: 标注清理函数

**修复位置**: `src/generator/book_finalizer.py`

**添加函数**:

```python
def clean_text(text):
    """清理文本中的标注和元数据"""
    if not text:
        return ""
    
    import re
    # 移除采访素材标注
    text = re.sub(r'采访素材 [^\n]*', '', text)
    
    # 移除来源标注
    text = re.sub(r'（来源：素材\d+）', '', text)
    text = re.sub(r'\(来源：素材\d+\)', '', text)
    
    # 移除证据不足标注
    text = re.sub(r'，其余细节尚无直接证据 [。.]?', '', text)
    text = re.sub(r'其余细节尚无直接证据', '', text)
    text = re.sub(r'采访素材仅能确认 [^\n]*', '', text)
    
    # 移除待核实标注
    text = re.sub(r'\[待核实\]', '', text)
    text = re.sub(r'待核实', '', text)
    
    # 移除事实核查标注
    text = re.sub(r'事实核查：[^\n]*', '', text)
    
    # 移除多余空格
    text = re.sub(r'  +', ' ', text)
    
    # 移除空行
    text = re.sub(r'\n\s*\n\s*\n', '\n\n', text)
    
    return text.strip()
```
