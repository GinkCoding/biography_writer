```python
def clean_text(text):
    """清理标注，只保留正文"""
    import re
    if not text:
        return ""
    
    # 移除采访素材标注
    text = re.sub(r'采访素材 [^\n]*', '', text)
    # 移除来源标注
    text = re.sub(r'（来源：素材\d+）', '', text)
    # 移除证据不足标注
    text = re.sub(r'，其余细节尚无直接证据 [。.]?', '', text)
    text = re.sub(r'其余细节尚无直接证据', '', text)
    # 清理多余空格
    text = re.sub(r' +', ' ', text)
    
    return text.strip()
```
